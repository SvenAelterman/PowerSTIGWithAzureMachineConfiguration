[CmdletBinding()]

$MetaData = Get-Content -Path './metadata.jsonc' | ConvertFrom-Json

[string]$Version = $MetaData.version

[hashtable]$params = @{
    Name          = "${{PolicyName}}-$Version"
    Configuration = './this/localhost.mof'
    Version       = $Version
    Type          = 'AuditAndSet'
    Force         = $true
    Path          = './Package'
}
New-GuestConfigurationPackage @params
